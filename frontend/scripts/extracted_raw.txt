          if (!seenCodes.has(item.code)) {
            seenCodes.add(item.code);
            const sameCode = result.results.filter(r => r.code === item.code);
            groupRows.push({
              code: item.code, name: item.name, teachers: sameCode,
              expanded: true, loadingReviews: true,
              teacherReviews: new Map(), teacherCredibility: new Map(),
            });
          }
        }
      } else {
        for (const [code, teachers] of codeMap) {
          groupRows.push({
            code, name: teachers[0].name, teachers,
            expanded: false, loadingReviews: false,
            teacherReviews: new Map(), teacherCredibility: new Map(),
          });
        }
        groupRows.sort((a, b) => a.code.localeCompare(b.code));
      }
      setGroups(groupRows);
      setStatus('success');
      // 加载评论
      setTimeout(() => loadReviewsForAll(groupRows), 100);
    } catch {
      setStatus('error'); setErrorMsg('搜索出错，请稍后重试');
    }
  }, []);

  const loadReviewsForAll = async (groupRows: GroupedRow[]) => {
    for (let idx = 0; idx < groupRows.length; idx++) {
      const row = groupRows[idx];
      if (row.isTeacherOverview) {
        for (const t of row.teachers) {
          try {
            const reviews = await getAllCourseReviews(t.id);
            row.teacherReviews.set(t.id, reviews);
            if (reviews.length > 0) {
              const cred = calculateCredibility(reviews);
              row.teacherCredibility.set(t.id, cred);
            }
          } catch {}
        }
      } else {
        for (const t of row.teachers) {
          if (!row.teacherReviews.has(t.id)) {
            try {
              const reviews = await getAllCourseReviews(t.id);
              row.teacherReviews.set(t.id, reviews);
              if (reviews.length > 0) {
                const cred = calculateCredibility(reviews);
                row.teacherCredibility.set(t.id, cred);
              }
            } catch {}
          }
        }
      }
      row.loadingReviews = false;
    }
    setGroups([...groupRows]);
  };

  const toggleGroup = (index: number) => {
    const newGroups = [...groups];
    const row = newGroups[index];
    if (!row.expanded) {
      row.expanded = true;
      if (row.teacherReviews.size === 0) {
        row.loadingReviews = true;
        setGroups(newGroups);
        loadReviewsForAll(newGroups);
      } else {
        setGroups(newGroups);
      }
    } else {
      row.expanded = false;
      setGroups(newGroups);
    }
  };

  const toggleTeacherReviews = (teacherKey: string) => {
    setExpandedTeachers(prev => {
      const newSet = new Set(prev);
      if (newSet.has(teacherKey)) newSet.delete(teacherKey);
      else newSet.add(teacherKey);
      return newSet;
    });
  };

  const [expandedTeachers, setExpandedTeachers] = useState<Set<string>>(new Set());

  const calcOverviewCredibility = (item: GroupedRow): AggregatedTeacherInfo | null => {
    if (!item.isTeacherOverview) return null;
    const allCreds: { avg: number; count: number }[] = [];
    for (const [, cred] of item.teacherCredibility) {
      allCreds.push(cred);
    }
    if (allCreds.length === 0) return null;
    const totalReviews = allCreds.reduce((s, c) => s + c.count, 0);
    const weighted = allCreds.reduce((s, c) => s + c.avg * c.count, 0) / totalReviews;
    const level = weighted >= 4 ? '高' : weighted >= 3 ? '中' : '低';
    return { avg: weighted, totalReviews, level };
  };

  const toggleFavorite = async (courseId: number) => {
    const isFav = favoriteIds.has(courseId);
    if (isFav) {
      await removeCommunityFavorite(courseId);
      favoriteIds.delete(courseId);
      setFavoriteIds(new Set(favoriteIds));
    } else {
      const info = { id: courseId, code: '', name: '', teacher: '' };
      await addCommunityFavorite(info);
      favoriteIds.add(courseId);
      setFavoriteIds(new Set(favoriteIds));
    }
  };

  useFocusEffect(useCallback(() => {
    (async () => {
      const favs: CommunityFavorite[] = await getCommunityFavorites();
      setFavoriteIds(new Set(favs.map(f => f.id)));
    })();
  }, []));

  const renderGroup = ({ item, index }: { item: GroupedRow; index: number }) => (
    <View style={styles.group}>
      <TouchableOpacity style={[styles.groupHeader, item.isTeacherOverview && styles.groupHeaderTeacher]} onPress={() => toggleGroup(index)} activeOpacity={0.7}>
        <View style={styles.groupHeaderLeft}>
          <MaterialIcons name={item.expanded ? 'expand-less' : 'expand-more'} size={20} color={item.isTeacherOverview ? C.primary : C.textSec} />
          {item.isTeacherOverview ? <MaterialIcons name="person" size={18} color={C.primary} /> : null}
          <Text style={[styles.code, item.isTeacherOverview && { color: C.credH }]}>
            {item.isTeacherOverview ? '综合' : item.code}
          </Text>
          {!item.isTeacherOverview && item.teachers[0]?.categories?.length > 0 ? (
            <View style={{ backgroundColor: getCategoryColor(item.teachers[0].categories[0]), borderRadius: 4, paddingHorizontal: 5, paddingVertical: 1, marginLeft: 6 }}>
              <Text style={{ color: '#FFF', fontSize: 10, fontWeight: '600' }}>{item.teachers[0].categories[0]}</Text>
            </View>
          ) : null}
          <Text style={styles.name} numberOfLines={1}>{item.name}</Text>
        </View>
        <Text style={styles.teacherCount}>
          {item.isTeacherOverview ? `${item.teachers.length}门课` : `${item.teachers.length}位老师`}
        </Text>
      </TouchableOpacity>

      {item.expanded && (
        <View style={styles.teacherList}>
          {item.loadingReviews && item.teacherReviews.size === 0 && (
            <ActivityIndicator size="small" color={C.primary} style={{ margin: 12 }} />
          )}

          {item.isTeacherOverview && (() => {
            const overview = calcOverviewCredibility(item);
            return overview ? (
              <View style={styles.teacherOverviewBanner}>
                <View style={styles.overviewRow}>
                  <MaterialIcons name="stars" size={28} color={C.star} />
                  <View style={{ flex: 1, marginLeft: 10 }}>
                    <Text style={styles.overviewTitle}>综合评分</Text>
                    <Text style={styles.overviewRating}>{ratingToStars(overview.avg)}  {overview.avg.toFixed(2)}</Text>
                    <Text style={styles.overviewSub}>共 {overview.totalReviews} 条点评 · {item.teachers.length} 门课</Text>
                  </View>
                  <View style={[styles.credBadge, {
                    backgroundColor: (overview.level === '高' ? C.credH : overview.level === '中' ? C.credM : C.credL) + '20',
                    borderColor: overview.level === '高' ? C.credH : overview.level === '中' ? C.credM : C.credL,
                  }]}>
                    <Text style={[styles.credText, { color: overview.level === '高' ? C.credH : overview.level === '中' ? C.credM : C.credL }]}>
                      可信度:{overview.level}
                    </Text>
                  </View>
                </View>
                <Text style={styles.overviewDivider}>各课程评分</Text>
              </View>
            ) : null;
          })()}

          {item.teachers.map(t => {
            const reviews = item.teacherReviews.get(t.id);
            const cred = item.teacherCredibility.get(t.id);
            const teacherKey = `${item.code}_${t.id}`;
            const showReviews = expandedTeachers.has(teacherKey);
            return (
              <TouchableOpacity key={t.id} style={styles.teacherCard} onPress={() => { if (reviews) toggleTeacherReviews(teacherKey); }} activeOpacity={0.7}>
                <View style={styles.teacherRow}>
                  {item.isTeacherOverview && <Text style={styles.courseCodeSmall}>{t.code}</Text>}
                  <View style={{ flex: 1 }}>
                    <Text style={styles.teacherName}>{item.isTeacherOverview ? t.name : t.teacher}</Text>
                    {reviews && reviews.length > 0 ? (
                      <View style={{ flexDirection: 'row', alignItems: 'center', marginTop: 2 }}>
                        <Text style={styles.teacherRating}>{ratingToStars(cred ? cred.avg : 0)}</Text>
                        <Text style={styles.teacherReviewsCount}>{reviews.length} 条</Text>
                      </View>
                    ) : item.loadingReviews ? (
                      <Text style={styles.csub}>加载中...</Text>
                    ) : (
                      <Text style={styles.csub}>暂无点评</Text>
                    )}
                  </View>
                  {reviews && reviews.length > 0 && (
                    <MaterialIcons name={showReviews ? 'expand-less' : 'expand-more'} size={20} color={C.textSec} />
                  )}
                  <TouchableOpacity onPress={() => toggleFavorite(t.id)} style={{ paddingLeft: 8 }}>
                    <MaterialIcons name={favoriteIds.has(t.id) ? 'favorite' : 'favorite-border'} size={20} color={favoriteIds.has(t.id) ? '#E53935' : '#CCC'} />
                  </TouchableOpacity>
                </View>

                {showReviews && reviews && (
                  <View style={styles.reviewsContainer}>
                    {reviews.slice(0, 5).map(r => (
                      <View key={r.id} style={styles.reviewCard}>
                        <View style={styles.reviewHeader}>
                          <Text style={styles.reviewSemester}>{r.semester.replace(/-/g, ' ')}</Text>
                          <Text style={styles.reviewRating}>{ratingToStars(r.rating)}</Text>
                        </View>
                        <Markdown style={mdStyles}>{r.comment}</Markdown>
                      </View>
                    ))}
                    {reviews.length > 5 && <Text style={styles.moreReviews}>还有 {reviews.length - 5} 条点评</Text>}
                  </View>
                )}
              </TouchableOpacity>
            );
          })}
        </View>
      )}
    </View>
  );

  return (
    <View style={[styles.container, { paddingTop: insets.top }]}>
      <View style={styles.header}>
        <TouchableOpacity style={styles.backBtn} onPress={() => navigation.goBack()} activeOpacity={0.7}>
          <MaterialIcons name="arrow-back" size={24} color={C.text} />
        </TouchableOpacity>
        <Text style={styles.headerTitle}>选课社区</Text>
        <View style={{ width: 40 }} />
      </View>

      <View style={styles.searchBar}>
        <MaterialIcons name="search" size={20} color={C.textSec} style={{ marginRight: 8 }} />
        <TextInput
          style={styles.searchInput}
          placeholder="搜索课程名称/课号/教师姓名"
          placeholderTextColor={C.textSec}
          value={searchText}
          onChangeText={text => {
            setSearchText(text);
            if (debounceRef.current) clearTimeout(debounceRef.current);
            debounceRef.current = setTimeout(() => handleSearch(text), 300);
          }}
          returnKeyType="search"
          autoCorrect={false}
        />
        {searchText.length > 0 && (
          <TouchableOpacity onPress={() => { setSearchText(''); setStatus('idle'); setGroups([]); }}>
            <MaterialIcons name="close" size={18} color={C.textSec} />
          </TouchableOpacity>
        )}
      </View>

      {status === 'idle' && (
        <View style={styles.emptyState}>
          <MaterialIcons name="forum" size={48} color="#DDD" />
          <Text style={styles.emptyText}>搜索课程名称查看不同老师的评分对比</Text>
          <Text style={styles.emptyHint}>支持模糊搜索，例如"数学分析"、"通选"</Text>
        </View>
      )}

      {status === 'loading' && (
        <View style={styles.loadingState}>
          <ActivityIndicator size="large" color={C.primary} />
          <Text style={styles.loadingText}>正在搜索...</Text>
        </View>
      )}

      {status === 'error' && (
        <View style={styles.emptyState}>
          <MaterialIcons name="search-off" size={48} color="#DDD" />
          <Text style={styles.emptyText}>{errorMsg}</Text>
        </View>
      )}

      {status === 'success' && (
        <>
          <FlatList
            ref={flatListRef}
            data={groups}
            renderItem={renderGroup}
            keyExtractor={item => item.code}
            contentContainerStyle={{ paddingBottom: 80 }}
            showsVerticalScrollIndicator={false}
            onScroll={Animated.event([{ nativeEvent: { contentOffset: { y: scrollY } } }], { useNativeDriver: false })}
            scrollEventThrottle={16}
            ListHeaderComponent={
              <Text style={styles.resultCount}>
                {(() => {
                  const uniqueTeachers = new Set<string>();
                  const realGroups = groups.filter(g => !g.isTeacherOverview);
                  for (const g of realGroups) for (const t of g.teachers) uniqueTeachers.add(t.teacher);
                  return `找到 ${uniqueTeachers.size} 位教师 · ${realGroups.length} 门课`;
                })()}
              </Text>
            }
          />
          {showScrollTop && (
            <TouchableOpacity style={styles.scrollTopBtn} onPress={() => flatListRef.current?.scrollToOffset({ offset: 0, animated: true })} activeOpacity={0.8}>
              <MaterialIcons name="keyboard-arrow-up" size={28} color="#FFF" />
            </TouchableOpacity>
          )}
        </>
      )}
    </View>
  );
};
